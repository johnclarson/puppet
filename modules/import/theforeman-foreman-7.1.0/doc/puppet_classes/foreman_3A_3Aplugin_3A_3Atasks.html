class foreman::plugin::tasks(
  $package = $foreman::plugin::tasks::params::package,
  $service = $foreman::plugin::tasks::params::service,
) inherits foreman::plugin::tasks::params {
  foreman::plugin { 'tasks':
    package => $package,
  } ~>
  service { 'foreman-tasks':
    ensure => running,
    enable => true,
    name   => $service,
  }
}
